CREATE TABLE CLIENTES
(NROCLI     INTEGER     NOT NULL,
 NYAPE      VARCHAR(20) NOT NULL,
 DOMICILIO  VARCHAR(30),
 LOCALIDAD  VARCHAR(20),
 SALDOCLI   DECIMAL(9,2) DEFAULT 0);
ALTER TABLE Clientes ADD CONSTRAINT Pk_Clientes PRIMARY KEY(nrocli);
-----------------------------------------------
CREATE TABLE RUBROS
(COD_RUBRO        INTEGER,
 DESCRIPCION      VARCHAR(30),
 CONSTRAINT PK_RUBROS PRIMARY KEY (COD_RUBRO) );
-----------------------------------------------
CREATE TABLE ARTICULOS
(NROARTIC         INTEGER NOT NULL,
 DESCRIPCION      VARCHAR(30) NOT NULL,
 RUBRO            INTEGER,
 STOCK            INTEGER NOT NULL DEFAULT 0 ,
 PTO_REPOSICION   INTEGER NOT NULL DEFAULT 0 ,
 PRECIO           DECIMAL(8,2) DEFAULT 0 ,
 CONSTRAINT PK_ARTICULOS PRIMARY KEY (NROARTIC),
 CONSTRAINT FK_ARTICULOS_RUBROS FOREIGN KEY (RUBRO) REFERENCES RUBROS(COD_RUBRO) ON DELETE NO ACTION);
------------------------------------------
CREATE TABLE FACTURAS
(NROFACTURA  INTEGER NOT NULL,
 CLIENTE     INTEGER NOT NULL,
 FECHA       TIMESTAMP NOT NULL DEFAULT NOW(),
 CONSTRAINT PK_FACTURAS PRIMARY KEY (NROFACTURA),
 CONSTRAINT FK_FACTURAS_CLIENTES FOREIGN KEY (CLIENTE) REFERENCES CLIENTES(NROCLI)
 ON DELETE NO ACTION
 ON UPDATE CASCADE);

------------------------------------------------------------
-- PRECIOUNI SIN "NOT NULL" Y SIN DEFAULT... SINO PROBLEMAS EN UPDATE

-- EJERCICIO E
select ar.*, count(de.cantidad) as cant from Articulos ar, Detalles de, facturas fa
where de.Nrofactura = fa.Nrofactura
and ar.Nroartic = de.articulo
and date_part ('year',fa.fecha) =2010
and date_part ('month',fa.fecha) =03
group by ar.nroartic
order by cant desc

-- EJERCICIO F
select fa.fecha, SUM(de.cantidad*de.preciouni) as total from detalles de, facturas fa
where de.nrofactura = fa.nrofactura
and date_part('month',fa.fecha) =04
and date_part('year',fa.fecha) =2010
group by fa.fecha
order by total asc

where de.nrofactura = fa.nrofactura
and date_part('month',fa.fecha) =04
and date_part('year',fa.fecha) =2010
group by fa.fecha
having SUM(de.cantidad*de.preciouni) >= 200
order by total asc

-- E) Averiguar la cantidad de cada uno de los artículos vendidos durante
--    marzo del 2010, ordenado segun la cantidad vendida.
--    (Rsta: 13 FILAS)

select distinct ar.*, SUM(de.cantidad) as cant from articulos ar, detalles de, facturas fa
where de.nrofactura = fa.nrofactura
and de.articulo = ar.nroartic
and date_part('year', fa.fecha) = 2010
and date_part('month', fa.fecha) = 03
group by ar.nroartic
order by cant desc

select fecha, ar.rubro from articulos ar, detalles de, facturas fa
where ar.nroartic = de.articulo
and de.nrofactura = fa.nrofactura
and (fa.fecha between '2010-04-15' and '2010-04-30')
group by ar.rubro, fa.fecha

-- J) Listar Numero y nombre de los cliente. Además, listar la cantidad de facturas que
--    tuvo durante Marzo y Abril del 2010 considerando SOLO las facturas en las que compro
--    articulos del rubro 3, ordenado por numero de cliente. (Rsta: 7 filas)


select cl.nrocli, cl.nyape, COUNT(fa.nrofactura) as facturas from  clientes cl, facturas fa,detalles de, articulos ar
where cl.nrocli = fa.cliente
and fa.nrofactura = de.nrofactura
and de.articulo = ar.nroartic
and (fa.fecha between '2010-03-01' and '2010-04-30')
and ar.rubro = 3
group by cl.nrocli, cl.nyape
order by facturas desc


-- K) Listar Numero y nombre de los cliente y la cantidad de facturas que tuvo durante
--    Marzo y Abril del 2010 con articulos del rubro 3 y CON MAS DE 2 FACTURAS, 
--    ordenado por numero de cliente. (Rsta: 1 fila)

select cl.nrocli, cl.nyape, COUNT(fa.nrofactura) as facturas from  clientes cl, facturas fa,detalles de, articulos ar
where cl.nrocli = fa.cliente
and fa.nrofactura = de.nrofactura
and de.articulo = ar.nroartic
and (fa.fecha between '2010-03-01' and '2010-04-30')
and ar.rubro = 3
group by cl.nrocli, cl.nyape
HAVING COUNT(fa.nrofactura) > 2
order by facturas desc


-- L) Calcular la cantidad de unidades del artículo 4040 vendidas en marzo del 2010.
--    (Rsta: 8 articulos)

select SUM(de.cantidad) as Vendidos from detalles de, facturas fa 
where de.nrofactura = fa.nrofactura
and date_part('year', fa.fecha) = 2010
and date_part('month', fa.fecha) = 03
and de.articulo = 4040

-- M) Calcular la cantidad de facturas en las que vendieron artículos del rubro 3
--    (Rsta: 26 facturas) 
select COUNT(fa.nrofactura) as NroFacturas from facturas fa, detalles de, articulos ar
where fa.nrofactura = de.nrofactura
and de.articulo = ar.nroartic
and ar.rubro = 3
group by fa.nrofactura
HAVING COUNT(fa.nrofactura) > 1
